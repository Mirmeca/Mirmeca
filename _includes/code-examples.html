Get your latest posts

PostsGateway(endpoint: "posts", env: nil).request({ (value: AnyObject?, error: NSError?) -> Void in
  if (error != nil) {
    println(error)
  } else {
    posts = value as! [Post]
  }
})

Display some of the post's infos

titleLabel.text = post.title!
subTitleLabel.text = post.author!.firstName!

Get a post's comments

CommentsGateway(endpoint: "posts/\(post.id!)/comments", env: nil).request({ (value: AnyObject?, error: NSError?) -> Void in
  if (error != nil) {
    println(error)
  } else {
    comments = value as! [Comment]
  }
})

Display a comment

nameLabel.text = comment.author!.name!
contentLabel.text = comment.content!
